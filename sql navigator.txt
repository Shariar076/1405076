--Aggregate Functions

SELECT COUNT(DISTINCT DEPARTMENT_ID) from EMPLOYEES;

SELECT avg(salary) from EMPLOYEES where DEPARTMENT_ID=20;

SELECT DEPARTMENT_ID,avg(salary) from EMPLOYEES GROUP BY DEPARTMENT_Id;

SELECT JOB_ID,DEPARTMENT_ID,avg(salary) from EMPLOYEES GROUP BY DEPARTMENT_Id,JOB_ID;

SELECT JOB_ID,DEPARTMENT_ID,avg(salary) from EMPLOYEES WHERE DEPARTMENT_ID=110 GROUP BY DEPARTMENT_Id,JOB_ID;

SELECT JOB_ID,DEPARTMENT_ID,avg(salary) from EMPLOYEES WHERE DEPARTMENT_ID=110 GROUP BY DEPARTMENT_Id,JOB_ID having avg(SALARY)>9000; --having written after group by
